01 Non-negotiables — cannot be cut for speed

P1 is the architectural baseline. Cutting any of these means Phase 2 requires a full rewrite, not an extension.

Pillar 1
Dynamic Field Registry live from Day 1. All 3 master forms render from DB. No hardcoded form fields anywhere.
Pillar 2
RabbitMQ Event Bus routes all inter-module events. Modules cannot call each other directly. Phase 2 features only subscribe to events.
Pillar 3
RBAC enforced server-side on every endpoint. Middleware, not business logic. Frontend checks are UX-only, never security.
Pillar 4
Append-only Audit Ledger active from Day 1. Every record change emits an event → Audit writes a revision row. Cannot retrofit.
Pillar 5
All labels bilingual from Day 1. label_en + label_hi on every field, node, status. Retrofitting Hindi breaks every table.
Pillar 6
Hierarchy as DB config. PS → District → HQ in hierarchy_nodes. Changeable from admin screen without deployment.
If any pillar is skipped "for speed" on Days 1–2, Phase 2 expansion breaks. The frontend can cut screens. The backend cannot cut architecture.

eventBus.js skeleton

// src/events/eventBus.js
const amqp = require('amqplib');
let channel = null;

async function connect() {
  const conn = await amqp.connect(process.env.RABBITMQ_URL);
  channel = await conn.createChannel();
  await channel.assertExchange('pharos', 'topic', { durable: true });
}

async function publish(routingKey, payload) {
  if (!channel) throw new Error('Event bus not connected');
  channel.publish('pharos', routingKey, Buffer.from(JSON.stringify(payload)), { persistent: true });
  console.log(`[EventBus] Published: ${routingKey}`);
}

async function subscribe(routingKey, handler) {
  const q = await channel.assertQueue('', { exclusive: true });
  await channel.bindQueue(q.queue, 'pharos', routingKey);
  channel.consume(q.queue, (msg) => {
    if (msg) {
      handler(JSON.parse(msg.content.toString()));
      channel.ack(msg);
    }
  });
}

module.exports = { connect, publish, subscribe };

api/utils.js — Axios instance with JWT interceptor (do this on Day 1)

// src/utils/api.js
import axios from 'axios';

const api = axios.create({ baseURL: import.meta.env.VITE_API_URL });

// Attach access token to every request
api.interceptors.request.use((config) => {
  const token = localStorage.getItem('access_token');
  if (token) config.headers.Authorization = `Bearer ${token}`;
  return config;
});

// Auto-refresh access token on 401
api.interceptors.response.use(
  (res) => res,
  async (err) => {
    if (err.response?.status === 401 && !err.config._retry) {
      err.config._retry = true;
      const refresh = localStorage.getItem('refresh_token');
      const { data } = await axios.post(`${import.meta.env.VITE_API_URL}/auth/refresh`, { refresh_token: refresh });
      localStorage.setItem('access_token', data.data.access_token);
      err.config.headers.Authorization = `Bearer ${data.data.access_token}`;
      return api(err.config);
    }
    return Promise.reject(err);
  }
);

export default api;

AuthContext — user state management

// src/contexts/AuthContext.jsx
import { createContext, useState, useContext } from 'react';
import { jwtDecode } from 'jwt-decode'; // npm install jwt-decode

const AuthContext = createContext(null);

export function AuthProvider({ children }) {
  const [user, setUser] = useState(() => {
    const token = localStorage.getItem('access_token');
    return token ? jwtDecode(token) : null;
  });

  function login(tokens) {
    localStorage.setItem('access_token', tokens.access_token);
    localStorage.setItem('refresh_token', tokens.refresh_token);
    setUser(jwtDecode(tokens.access_token));
  }

  function logout() {
    localStorage.clear();
    setUser(null);
  }

  return <AuthContext.Provider value={{ user, login, logout }}>{children}</AuthContext.Provider>;
}

export const useAuth = () => useContext(AuthContext);

10 DynamicForm — the core frontend component

This is the most important component. It reads field definitions from the registry and renders the correct input for each field type. Get this right on Day 2 and all 3 master forms are essentially done.
// src/components/forms/DynamicForm.jsx
import { Form, Input, Select, DatePicker, Upload, Checkbox, Row, Col, Button } from 'antd';
import { useQuery } from '@tanstack/react-query';
import { useTranslation } from 'react-i18next';
import api from '../../utils/api';

// Maps field_type from registry → Ant Design component
const COMPONENTS = {
  TEXT:        () => <Input />,
  NUMBER:      () => <Input type="number" />,
  DATE:        () => <DatePicker style={{ width: '100%' }} />,
  DATETIME:    () => <DatePicker showTime style={{ width: '100%' }} />,
  BOOLEAN:     () => <Checkbox />,
  PHONE:       () => <Input type="tel" />,
  TEXTAREA:    () => <Input.TextArea rows={3} />,
  FILE:        () => <Upload beforeUpload={() => false}><Button>Upload</Button></Upload>,
  SELECT:      (field, lang) => (
    <Select options={(field.options || []).map(o => ({ value: o.value, label: lang === 'hi' ? o.label_hi : o.label_en }))} />
  ),
  MULTISELECT: (field, lang) => (
    <Select mode="multiple" options={(field.options || []).map(o => ({ value: o.value, label: lang === 'hi' ? o.label_hi : o.label_en }))} />
  ),
};

export default function DynamicForm({ recordType, initialValues, onSaveDraft, onSubmit, readOnly = false }) {
  const [form] = Form.useForm();
  const { i18n } = useTranslation();
  const lang = i18n.language;

  const { data, isLoading } = useQuery({
    queryKey: ['fields', 'form', recordType],
    queryFn: () => api.get(`/fields/form/${recordType}`).then(r => r.data.data),
  });

  if (isLoading) return <div>Loading form...</div>;

  return (
    <Form form={form} layout="vertical" initialValues={initialValues?.data}
          onValuesChange={(_, all) => onSaveDraft?.({ data: all })}>
      {(data || []).map(section => (
        <div key={section.section} style={{ marginBottom: 24 }}>
          <h3 style={{ marginBottom: 12, color: '#1e293b' }}>
            {lang === 'hi' ? section.title_hi : section.title_en}
          </h3>
          <Row gutter={16}>
            {section.fields.map(field => {
              const label = lang === 'hi' ? field.label_hi : field.label_en;
              const rules = [];
              if (field.validation_rules?.required) rules.push({ required: true, message: `${label} is required` });
              const ComponentFn = COMPONENTS[field.field_type] || COMPONENTS.TEXT;
              return (
                <Col span={12} key={field.field_key}>
                  <Form.Item name={field.field_key} label={label} rules={rules}>
                    {ComponentFn(field, lang)}
                  </Form.Item>
                </Col>
              );
            })}
          </Row>
        </div>
      ))}
      {!readOnly && (
        <div style={{ display: 'flex', gap: 8 }}>
          <Button onClick={() => form.validateFields().then(vals => onSaveDraft?.({ data: vals }))}>Save Draft</Button>
          <Button type="primary" onClick={() => form.validateFields().then(vals => onSubmit?.({ data: vals }))}>Submit</Button>
        </div>
      )}
    </Form>
  );
}
